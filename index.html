15 Minutes Later


#!/bin/bash

function print_links {
  xmlstarlet sel -t -v '//ulink/@url' 2>/dev/null "$1" | \
  sort -u | sed '/^$/d'
}

function check_link {
  curl --connect-timeout 5 --retry 3 \
       -4ILfks "$1" &>/dev/null
}

function print_status {
  if check_link "$1"; then
    echo "[ PASS ]  $1"
  else
    echo "[ FAIL ]  $1"
  fi
}

export -f print_status check_link
print_links "$1" | \
  xargs -n 1 -P 0 bash -c 'print_status "$@"' --
